#### js原生复制文本
```js
/**
 * 点击获取值复制
 * @param {String} text 需要复制的值
 * @param {String} callBack 执行的回调
 */
export const copyText = (text, callBack) => {
  let input = document.createElement("input");
  //将input的值设置为需要复制的内容
  input.value = text;
  //添加input标签
  document.body.appendChild(input);
  //选中input标签
  input.select();
  //执行复制
  document.execCommand("copy");
  // 回调函数
  callBack && callBack();
  //移除input标签
  document.body.removeChild(input);
};
```
